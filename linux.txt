#1.Выведите список всех файлов в папке html 

  $ls var/www/html 						#Использую ls для вывода списка файлов в папке html


#2.Найдите в файле access.log все упоминания строки "404"

  $grep "404" var/logs/access.log    				#Использую команду grep для поиска строк с упоминанием "404" в файле access.log


#3.На основании файла 404.html сделайте файл 403.html заменив в его содержимом все упоминания 404 на 403 

  $cp var/www/html/404.html var/www/html/403.html 		#Командой cp создаю копию файла 404.html в том же каталоге, с именем 403.html
  $sed -i s/404/403/ var/www/html/403.html			#Командой sed заменяю все упоминания 404 на 403, опция -i для сохранения изменений, и s// для указания что заменять
  $cat var/www/html/403.html					#просмотрим новый файл


#4.Модифицируйте права доступа и владельца файла index.html следующим образом: Текущий пользователь должен иметь права на чтение/запись Группа www-data на чтение Все остальные не имеют доступа к файлу 

  $chmod u+rw- var/www/html/index.html				#добавили права на чтение и запись файла index.html для текущего владельца
  $chgrp www-data var/www/html/index.html			#добавили группу www-data
  $chmod g+r-- var/www/html/index.html				#добавили права на чтения файла index.html для группы
  $chmod o-rwx var/www/html/index.html				#отобрали все права у остальных пользователей, для файла index.html
  $ls -l var/www/html/index.html				#проверим изменения прав доступа


#5.Сделайте символическую ссылку на файл access.log в папке html. 

  $ln -s ../../logs/access.log var/www/html/lnk_access.log	#создали символическую ссылку (-s) lnk_access.log на файл access.log
  $cat var/www/html/lnk_access.log				#проверим содержимое

#6 Выведите на экран все файлы в папке var 

  $ls -R var							#Использую ls с флагом -R для вывода списка всех файлов в папке и подпапках var


#7.Создайте новый файл 503.html в папке html с следующим содержимым:
 
  $cat>var/www/html/503.html					#Создаю файл 503.html в каталоге html, и заполняю его через терминал, сохраняю командой ctrl+d.
  $<html>
  $ <head>
  $    <title>503</title>
  $ </head>
  $</html>
  $cat var/www/html/503.html					#просмотрим новый файл
